```tsx
import type {
  ICraftPlayerInstance,
} from "@icraft/player";
import { ICraftPlayer } from "@icraft/player";
import { useRef } from "react";

export default () => {
  const instanceRef = useRef<ICraftPlayerInstance>();

  return (
    <ICraftPlayer
      fileUrl="/docker.iplayer"
      ref={instanceRef}
      showAnimationPanel
    />
  );
};


```
